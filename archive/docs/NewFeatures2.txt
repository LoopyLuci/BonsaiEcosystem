#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
pub enum Error {
    /// The knowledge base returned no evidence for the claim.
    GroundingFailed(String),
    /// The formal verifier rejected the output.
    VerificationFailed(String),
    /// The bias detector flagged the output.
    BiasDetected(String),
    /// The model's confidence was below the required threshold.
    ConfidenceTooLow(f32, f32),
    /// The output was empty.
    EmptyOutput,
    /// A required capability was missing.
    MissingCapability(String),
    /// An internal service timed out.
    Timeout(String),
    /// The knowledge base contradicted the claim.
    Contradiction(String),
    /// A generic I/O error.
    IoError(String),
    /// Serialization failure.
    SerializationError(String),
}

impl fmt::Display for Error {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Error::GroundingFailed(msg) => write!(f, "Grounding failed: {}", msg),
            Error::VerificationFailed(msg) => write!(f, "Verification failed: {}", msg),
            Error::BiasDetected(msg) => write!(f, "Bias detected: {}", msg),
            Error::ConfidenceTooLow(got, need) => {
                write!(f, "Confidence too low: {:.2} < {:.2}", got, need)
            }
            Error::EmptyOutput => write!(f, "Empty output"),
            Error::MissingCapability(cap) => write!(f, "Missing capability: {}", cap),
            Error::Timeout(svc) => write!(f, "Timeout: {}", svc),
            Error::Contradiction(msg) => write!(f, "Contradiction: {}", msg),
            Error::IoError(msg) => write!(f, "I/O error: {}", msg),
            Error::SerializationError(msg) => write!(f, "Serialization error: {}", msg),
        }
    }
}

//  Claim – a single subject‑predicate‑object triple

#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub struct Claim {
    pub subject: String,
    pub predicate: String,
    pub object: String,
    /// Optional qualifiers (e.g., temporal, spatial).
    pub qualifiers: Vec<String>,
}

impl Claim {
    /// Create a new claim with the three core components.
    pub fn new(subject: &str, predicate: &str, object: &str) -> Self {
        Self {
            subject: subject.to_string(),
            predicate: predicate.to_string(),
            object: object.to_string(),
            qualifiers: vec![],
        }
    }

    /// Create a claim with additional qualifiers.
    pub fn with_qualifiers(
        subject: &str,
        predicate: &str,
        object: &str,
        qualifiers: Vec<String>,
    ) -> Self {
        Self {
            subject: subject.to_string(),
            predicate: predicate.to_string(),
            object: object.to_string(),
            qualifiers,
        }
    }

    /// Produce a canonical hash for content‑addressed lookup.
    pub fn normalised_hash(&self) -> String {
        let mut hasher = blake3::Hasher::new();
        hasher.update(self.subject.as_bytes());
        hasher.update(b"|");
        hasher.update(self.predicate.as_bytes());
        hasher.update(b"|");
        hasher.update(self.object.as_bytes());
        for q in &self.qualifiers {
            hasher.update(b"|");
            hasher.update(q.as_bytes());
        }
        hasher.finalize().to_hex().to_string()
    }

    /// Returns true if the claim is trivially empty (all fields blank).
    pub fn is_empty(&self) -> bool {
        self.subject.is_empty() && self.predicate.is_empty() && self.object.is_empty()
    }
}

impl fmt::Display for Claim {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "({}, {}, {})", self.subject, self.predicate, self.object)
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum GroundingScore {
    /// The claim was found with the associated evidence.
    Found(GroundingEvidence),
    /// The claim could not be located in any trusted source.
    NotFound,
    /// The claim was explicitly contradicted by a trusted source.
    Contradicted(GroundingEvidence),
}

impl GroundingScore {
    /// Returns true if the claim has positive evidence.
    pub fn is_grounded(&self) -> bool {
        matches!(self, GroundingScore::Found(_))
    }

    /// Returns the confidence from the evidence, or 0.0 if not found.
    pub fn confidence(&self) -> f32 {
        match self {
            GroundingScore::Found(ev) => ev.confidence,
            GroundingScore::Contradicted(ev) => ev.confidence,
            GroundingScore::NotFound => 0.0,
        }
    }
}

/// Extract subject‑predicate‑object claims from a natural language string.
///
/// This is a deterministic, rule‑based extractor that works without AI.
/// It uses simple dependency patterns to identify factual assertions.
pub fn extract_claims(text: &str) -> Vec<Claim> {
    let mut claims = Vec::new();

    // Split into sentences
    for sentence in text.split(|c| c == '.' || c == '!' || c == '?') {
        let sentence = sentence.trim();
        if sentence.is_empty() {
            continue;
        }

        // Pattern 1: "X is Y" or "X are Y"
        if let Some(claim) = extract_is_pattern(sentence) {
            claims.push(claim);
        }

        // Pattern 2: "X has Y" or "X have Y"
        if let Some(claim) = extract_has_pattern(sentence) {
            claims.push(claim);
        }

        // Pattern 3: "The capital of X is Y"
        if let Some(claim) = extract_capital_pattern(sentence) {
            claims.push(claim);
        }

        // Pattern 4: "X was described by Y"
        if let Some(claim) = extract_described_by_pattern(sentence) {
            claims.push(claim);
        }

        // Pattern 5: "X verb Y" (simple subject‑verb‑object)
        if let Some(claim) = extract_svo_pattern(sentence) {
            claims.push(claim);
        }
    }

    claims
}

fn extract_is_pattern(sentence: &str) -> Option<Claim> {
    let lower = sentence.to_lowercase();
    if let Some(pos) = lower.find(" is ") {
        let subject = sentence[..pos].trim();
        let object = sentence[pos + 4..].trim();
        if !subject.is_empty() && !object.is_empty() {
            return Some(Claim::new(subject, "is", object));
        }
    }
    if let Some(pos) = lower.find(" are ") {
        let subject = sentence[..pos].trim();
        let object = sentence[pos + 5..].trim();
        if !subject.is_empty() && !object.is_empty() {
            return Some(Claim::new(subject, "are", object));
        }
    }
    None
}

fn extract_has_pattern(sentence: &str) -> Option<Claim> {
    let lower = sentence.to_lowercase();
    if let Some(pos) = lower.find(" has ") {
        let subject = sentence[..pos].trim();
        let object = sentence[pos + 5..].trim();
        if !subject.is_empty() && !object.is_empty() {
            return Some(Claim::new(subject, "has", object));
        }
    }
    None
}

fn extract_capital_pattern(sentence: &str) -> Option<Claim> {
    let lower = sentence.to_lowercase();
    if let Some(pos) = lower.find("capital of ") {
        let rest = &sentence[pos + 11..];
        if let Some(is_pos) = rest.to_lowercase().find(" is ") {
            let subject = &rest[..is_pos].trim();
            let object = rest[is_pos + 4..].trim();
            let full_subject = format!("capital of {}", subject);
            return Some(Claim::new(&full_subject, "is", object));
        }
    }
    None
}

fn extract_described_by_pattern(sentence: &str) -> Option<Claim> {
    let lower = sentence.to_lowercase();
    if let Some(pos) = lower.find(" described by ") {
        let subject = sentence[..pos].trim();
        let object = sentence[pos + 14..].trim();
        if !subject.is_empty() && !object.is_empty() {
            return Some(Claim::new(subject, "described_by", object));
        }
    }
    None
}

fn extract_svo_pattern(sentence: &str) -> Option<Claim> {
    let words: Vec<&str> = sentence.split_whitespace().collect();
    if words.len() >= 3 {
        let subject = words[0];
        let verb = words[1];
        let object = words[2..].join(" ");
        // Only extract if the verb is a common relational verb
        if matches!(
            verb.to_lowercase().as_str(),
            "contains"
                | "includes"
                | "supports"
                | "requires"
                | "provides"
                | "implements"
                | "defines"
        ) {
            return Some(Claim::new(subject, verb, &object));
        }
    }
    None
}

// ---------------------------------------------------------------------------
//  Safety envelope application
// ---------------------------------------------------------------------------

/// Apply the safety envelope to a text, replacing blocked phrases and
/// adjusting certainty language.
pub fn apply_safety_envelope(text: &str, envelope: &SafetyEnvelope) -> String {
    let mut result = text.to_string();

    // Check for blocked phrases first
    for blocked in &envelope.blocked_phrases {
        if result.to_lowercase().contains(&blocked.to_lowercase()) {
            // Replace the blocked phrase with a hedge
            result = result.replace(blocked, "[statement requires verification]");
        }
    }

    // Apply phrase replacements
    for (original, replacement) in &envelope.phrase_replacements {
        result = result.replace(original, replacement);
    }

    result
}

// crates/ahf-formal-verifier/src/lib.rs
//! Formal Verifier – validates AI outputs against schemas, session history,
//! and logical constraints. Integrates with Axiom for proof checking.

use ahf_core::{Claim, Error, FormalVerificationResult, Result};
use serde::{Deserialize, Serialize};
use serde_json::Value;
use std::collections::{HashMap, HashSet};
use std::sync::Arc;
use tokio::sync::RwLock;

// ---------------------------------------------------------------------------
//  Schema definition
// ---------------------------------------------------------------------------

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct OutputSchema {
    /// Expected JSON Schema (if output is structured).
    pub json_schema: Option<Value>,
    /// Required fields (for structured output).
    pub required_fields: Vec<String>,
    /// Maximum allowed length in characters.
    pub max_length: usize,
    /// Allowed output formats.
    pub allowed_formats: Vec<String>,
    /// Logical constraints that must hold.
    pub constraints: Vec<Constraint>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Constraint {
    /// Human‑readable description.
    pub description: String,
    /// Axiom theorem name (if applicable).
    pub axiom_theorem: Option<String>,
    /// Simple predicate: field name, operator, value.
    pub predicate: Option<ConstraintPredicate>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ConstraintPredicate {
    pub field: String,
    pub operator: ConstraintOperator,
    pub value: serde_json::Value,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum ConstraintOperator {
    Equals,
    NotEquals,
    GreaterThan,
    LessThan,
    GreaterOrEqual,
    LessOrEqual,
    Contains,
    NotContains,
    InRange { min: f64, max: f64 },
}

// ---------------------------------------------------------------------------
//  Session History
// ---------------------------------------------------------------------------

#[derive(Debug, Clone)]
struct SessionEntry {
    claims: HashSet<String>, // normalised hashes of accepted claims
    timestamp: u64,
}

pub struct SessionHistory {
    entries: Vec<SessionEntry>,
    max_entries: usize,
}

impl SessionHistory {
    pub fn new(max_entries: usize) -> Self {
        Self {
            entries: Vec::new(),
            max_entries,
        }
    }

    /// Add a set of accepted claims to the session history.
    pub fn record(&mut self, claims: &[Claim]) {
        let hashes: HashSet<String> = claims.iter().map(|c| c.normalised_hash()).collect();
        self.entries.push(SessionEntry {
            claims: hashes,
            timestamp: 0, // would use monotonic clock
        });
        if self.entries.len() > self.max_entries {
            self.entries.remove(0);
        }
    }

    /// Check if a new claim contradicts any previously accepted claim.
    /// Contradiction is detected when the same subject‑predicate pair has
    /// a different object.
    pub fn check_contradiction(&self, claims: &[Claim]) -> Vec<String> {
        let mut violations = Vec::new();
        let all_previous: HashSet<String> = self
            .entries
            .iter()
            .flat_map(|e| e.claims.iter().cloned())
            .collect();

        for claim in claims {
            // For each claim, check if we previously accepted a different
            // claim with the same subject and predicate but different object.
            for prev_entry in &self.entries {
                for prev_hash in &prev_entry.claims {
                    // In a real implementation, we'd decode the hash or store
                    // the actual claim. For now, we compare the claim's own
                    // normalised hash – if it's not in the set, it's new and
                    // potentially contradictory with something that shares
                    // subject+predicate. A full implementation would store
                    // the claims themselves.
                    let _ = prev_hash;
                }
            }
            let _ = all_previous;
            let _ = claim;
        }

        violations
    }
}

// ---------------------------------------------------------------------------
//  Formal Verifier
// ---------------------------------------------------------------------------

pub struct FormalVerifier {
    /// Registered schemas by name.
    schemas: Arc<RwLock<HashMap<String, OutputSchema>>>,
    /// Per‑session history.
    sessions: Arc<RwLock<HashMap<String, SessionHistory>>>,
}

impl FormalVerifier {
    pub fn new() -> Self {
        Self {
            schemas: Arc::new(RwLock::new(HashMap::new())),
            sessions: Arc::new(RwLock::new(HashMap::new())),
        }
    }

    /// Register a schema that can be used for validation.
    pub async fn register_schema(&self, name: &str, schema: OutputSchema) {
        let mut schemas = self.schemas.write().await;
        schemas.insert(name.to_string(), schema);
    }

    /// Validate raw output against a named schema.
    pub async fn validate(
        &self,
        raw_output: &str,
        schema_name: &str,
        session_id: Option<&str>,
    ) -> Result<FormalVerificationResult> {
        let schemas = self.schemas.read().await;
        let schema = schemas
            .get(schema_name)
            .ok_or_else(|| Error::VerificationFailed(format!("Unknown schema: {}", schema_name)))?;

        let mut violations = Vec::new();

        // 1. Length check
        if raw_output.len() > schema.max_length {
            violations.push(format!(
                "Output exceeds maximum length: {} > {}",
                raw_output.len(),
                schema.max_length
            ));
        }

        // 2. JSON schema validation (if applicable)
        if let Some(ref json_schema) = schema.json_schema {
            if let Ok(parsed) = serde_json::from_str::<Value>(raw_output) {
                // Use a JSON schema validator (e.g., jsonschema crate)
                // For now, we do a simple required‑fields check.
                if let Some(obj) = parsed.as_object() {
                    for field in &schema.required_fields {
                        if !obj.contains_key(field) {
                            violations.push(format!("Missing required field: {}", field));
                        }
                    }
                }
            } else if schema.required_fields.is_empty() {
                // Not JSON, but no required fields – acceptable.
            } else {
                violations.push("Output is not valid JSON but schema requires structured output.".into());
            }
        }

        // 3. Constraint validation
        for constraint in &schema.constraints {
            if let Some(ref predicate) = constraint.predicate {
                if !self.check_predicate(raw_output, predicate) {
                    violations.push(constraint.description.clone());
                }
            }
        }

        // 4. Session consistency (if a session ID is provided)
        if let Some(sid) = session_id {
            let mut sessions = self.sessions.write().await;
            let history = sessions
                .entry(sid.to_string())
                .or_insert_with(|| SessionHistory::new(100));
            let claims = ahf_core::extract_claims(raw_output);
            let contradictions = history.check_contradiction(&claims);
            violations.extend(contradictions);
            // Record accepted claims for future consistency checks
            if violations.is_empty() {
                history.record(&claims);
            }
        }

        if violations.is_empty() {
            Ok(FormalVerificationResult::Valid)
        } else {
            Ok(FormalVerificationResult::Invalid(violations))
        }
    }

    fn check_predicate(&self, raw_output: &str, predicate: &ConstraintPredicate) -> bool {
        // Try to parse the output as JSON and extract the field value.
        let value = if let Ok(parsed) = serde_json::from_str::<Value>(raw_output) {
            parsed
                .get(&predicate.field)
                .cloned()
                .unwrap_or(Value::Null)
        } else {
            // For plain text, try to extract numeric values.
            extract_numeric_field(raw_output, &predicate.field)
        };

        match &predicate.operator {
            ConstraintOperator::Equals => value == predicate.value,
            ConstraintOperator::NotEquals => value != predicate.value,
            ConstraintOperator::GreaterThan => {
                if let (Some(a), Some(b)) = (value.as_f64(), predicate.value.as_f64()) {
                    a > b
                } else {
                    false
                }
            }
            ConstraintOperator::LessThan => {
                if let (Some(a), Some(b)) = (value.as_f64(), predicate.value.as_f64()) {
                    a < b
                } else {
                    false
                }
            }
            ConstraintOperator::GreaterOrEqual => {
                if let (Some(a), Some(b)) = (value.as_f64(), predicate.value.as_f64()) {
                    a >= b
                } else {
                    false
                }
            }
            ConstraintOperator::LessOrEqual => {
                if let (Some(a), Some(b)) = (value.as_f64(), predicate.value.as_f64()) {
                    a <= b
                } else {
                    false
                }
            }
            ConstraintOperator::Contains => {
                if let (Some(s), Some(sub)) = (value.as_str(), predicate.value.as_str()) {
                    s.contains(sub)
                } else {
                    false
                }
            }
            ConstraintOperator::NotContains => {
                if let (Some(s), Some(sub)) = (value.as_str(), predicate.value.as_str()) {
                    !s.contains(sub)
                } else {
                    true
                }
            }
            ConstraintOperator::InRange { min, max } => {
                if let Some(v) = value.as_f64() {
                    v >= *min && v <= *max
                } else {
                    false
                }
            }
        }
    }
}

/// Try to extract a numeric value from plain text by field name.
fn extract_numeric_field(text: &str, field_name: &str) -> Value {
    let lower = text.to_lowercase();
    let field_lower = field_name.to_lowercase();

    // Look for patterns like "field_name: 123" or "field_name = 123"
    for pattern in &[
        format!("{}:", field_lower),
        format!("{} =", field_lower),
        format!("{} is", field_lower),
    ] {
        if let Some(pos) = lower.find(&pattern) {
            let rest = &text[pos + pattern.len()..];
            if let Some(num_str) = rest.split_whitespace().next() {
                if let Ok(num) = num_str.parse::<f64>() {
                    return Value::Number(
                        serde_json::Number::from_f64(num).unwrap_or(serde_json::Number::from(0)),
                    );
                }
            }
        }
    }
    Value::Null
}

//! bias detection, and model confidence to accept, reject, or escalate

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ArbiterConfig {

    pub grounding_threshold: f64,

            confidence_threshold: 0.8,

    }

// ---------------------------------------------------------------------------
//  Policy – versioned, council‑signed configuration
// ---------------------------------------------------------------------------

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Policy {
    pub version: String,
    pub config: ArbiterConfig,
    pub safety_envelope: SafetyEnvelope,
    pub fallback_responses: Vec<FallbackResponse>,
    /// BLS signature from the governance council.
    pub council_signature: Option<Vec<u8>>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct FallbackResponse {
    pub pattern: String,
    pub response: String,
}

impl Policy {
    pub fn default_policy() -> Self {
        Self {
            version: "1.0.0".into(),
            config: ArbiterConfig::default(),
            safety_envelope: SafetyEnvelope::default(),
            fallback_responses: vec![
                FallbackResponse {
                    pattern: "grounding".into(),
                    response:
                        "I cannot provide a verified answer to that question based on available knowledge sources."
                            .into(),
                },
                FallbackResponse {
                    pattern: "confidence".into(),
                    response: "I'm not confident enough in this response to share it. Could you rephrase or ask a different question?"
                        .into(),
                },
                FallbackResponse {
                    pattern: "bias".into(),
                    response: "I cannot provide an answer that may contain biased or harmful content."
                        .into(),
                },
                FallbackResponse {
                    pattern: "contradiction".into(),
                    response: "I've detected a contradiction with previously established facts. I need to verify before responding."
                        .into(),
                },
            ],
            council_signature: None,
        }
    }

    pub fn find_fallback(&self, reason_type: &str) -> String {
        self.fallback_responses
            .iter()
            .find(|f| reason_type.contains(&f.pattern))
            .map(|f| f.response.clone())
            .unwrap_or_else(|| {
                "I cannot provide a verified answer at this time.".into()
            })
    }
}

// ---------------------------------------------------------------------------
//  Arbiter
// ---------------------------------------------------------------------------

pub struct Arbiter {
    policy: Arc<RwLock<Policy>>,
}

impl Arbiter {
    pub fn new() -> Self {
        Self {
            policy: Arc::new(RwLock::new(Policy::default_policy())),
        }
    }

    pub fn with_policy(policy: Policy) -> Self {
        Self {
            policy: Arc::new(RwLock::new(policy)),
        }
    }

    /// Update the active policy (hot‑reload).
    pub async fn update_policy(&self, new_policy: Policy) {
        let mut policy = self.policy.write().await;
        *policy = new_policy;
    }

    /// Make an acceptance decision.
    ///
    /// The decision follows these rules in precedence order:
    /// 1. If formal verification failed → reject.
    /// 2. If grounding score < threshold → reject.
    /// 3. If bias score > threshold → reject.
    /// 4. If model confidence < threshold → reject.
    /// 5. If critical and grounding not perfect → escalate.
    /// 6. Otherwise → accept with safety envelope.
    pub async fn decide(
        &self,
        grounding_score: f64,
        verification: &FormalVerificationResult,
        bias_score: f32,
        model_confidence: f32,
    ) -> ArbiterDecision {
        let policy = self.policy.read().await;

        // Rule 1: Formal verification failure → immediate rejection
        if !verification.is_valid() {
            let violations = match verification {
                FormalVerificationResult::Invalid(v) => v.join(", "),
                _ => String::new(),
            };
            return ArbiterDecision {
                accepted: false,
                reason: format!("Formal verification failed: {}", violations),
                safety_envelope: policy.safety_envelope.clone(),
                fallback: policy.find_fallback("verification"),
            };
        }

        // Rule 2: Insufficient grounding
        if grounding_score < policy.config.grounding_threshold {
            return ArbiterDecision {
                accepted: false,
                reason: format!(
                    "Grounding score {:.2} below threshold {:.2}",
                    grounding_score, policy.config.grounding_threshold
                ),
                safety_envelope: policy.safety_envelope.clone(),
                fallback: policy.find_fallback("grounding"),
            };
        }

        // Rule 3: Bias detected
        if bias_score > policy.config.bias_threshold {
            return ArbiterDecision {
                accepted: false,
                reason: format!(
                    "Bias score {:.2} exceeds threshold {:.2}",
                    bias_score, policy.config.bias_threshold
                ),
                safety_envelope: policy.safety_envelope.clone(),
                fallback: policy.find_fallback("bias"),
            };
        }

        // Rule 4: Low model confidence
        if model_confidence < policy.config.confidence_threshold {
            return ArbiterDecision {
                accepted: false,
                reason: format!(
                    "Model confidence {:.2} below threshold {:.2}",
                    model_confidence, policy.config.confidence_threshold
                ),
                safety_envelope: policy.safety_envelope.clone(),
                fallback: policy.find_fallback("confidence"),
            };
        }

        // Rule 5: Accepted – apply safety envelope
        ArbiterDecision {
            accepted: true,
            reason: "All checks passed".into(),
            safety_envelope: policy.safety_envelope.clone(),
            fallback: String::new(),
        }
    }

    /// Get the current safety envelope (for applying to accepted outputs).
    pub async fn safety_envelope(&self) -> SafetyEnvelope {
        self.policy.read().await.safety_envelope.clone()
    }
}

                regex: r"(?i)\b(poor|homeless)\s+people\s+are\s+(lazy|just|simply)\b".into(),
                category: BiasCategory::Socioeconomic,
                severity: 0.8,
                explanation: "Makes negative assumptions about people based on economic status.".into(),

                name: "nationality_derogatory".into(),
                regex: r"(?i)\b(all|most)\s+(french|british|american|chinese|indian|german|russian)\s+are\s+(rude|arrogant|stupid|lazy)\b".into(),
                category: BiasCategory::Nationality,
                severity: 0.85,
                explanation: "Makes sweeping negative claims about a nationality.".into(),
            },
        ];

        // Compile regexes synchronously for the initial load
        let compiled: Vec<(Regex, BiasPattern)> = defaults
            .into_iter()
            .filter_map(|p| Regex::new(&p.regex).ok().map(|r| (r, p)))
            .collect();

        // Since we can't block on async here, we use try_write or block.
        // In production, this would be done at startup.
        futures::executor::block_on(async {
            let mut patterns = self.patterns.write().await;
            *patterns = compiled;
        });
    }

        let regex = Regex::new(&pattern.regex)
            .map_err(|e| Error::SerializationError(format!("Invalid regex: {}", e)))?;
        let mut patterns = self.patterns.write().await;
        patterns.push((regex, pattern));
        Ok(())
    }

    /// Detect bias in a text. Returns a report with overall score and
    /// matched patterns.
    pub async fn detect(&self, text: &str) -> BiasReport {
        let patterns = self.patterns.read().await;
        let mut matched_patterns = Vec::new();
        let mut total_severity = 0.0f32;
        let mut max_severity = 0.0f32;
        let mut explanations = Vec::new();

        for (regex, pattern) in patterns.iter() {
            if regex.is_match(text) {
                matched_patterns.push(pattern.name.clone());
                total_severity += pattern.severity;
                max_severity = max_severity.max(pattern.severity);
                explanations.push(format!("{}: {}", pattern.name, pattern.explanation));
            }
        }

        // Combine with ML classifier if available
        let ml_flagged = if let Some(ref classifier) = self.ml_classifier {
            let ml_score = classifier.classify(text).await.unwrap_or(0.0);
            if ml_score > 0.5 {
                total_severity = total_severity.max(ml_score);
                max_severity = max_severity.max(ml_score);
                true
            } else {
                false
            }
        } else {
            false
        };

        // Compute overall bias score: weighted by max severity, capped at 1.0.
        let bias_score = if matched_patterns.is_empty() && !ml_flagged {
            0.0
        } else {
            (max_severity * 0.7 + (matched_patterns.len() as f32 * 0.1).min(0.3)).min(1.0)
        };

        // Update statistics
        {
            let mut stats = self.stats.write().await;
            stats.total_checks += 1;
            if bias_score > 0.5 {
                stats.total_flagged += 1;
            }
        }

        BiasReport {
            bias_score,
            matched_patterns,
            ml_flagged,
            explanation: explanations.join("; "),
        }
    }

    /// Get current bias detection statistics.
    pub async fn stats(&self) -> BiasStats {
        self.stats.read().await.clone()
    }
}

//! AHF Gateway – the main orchestrator that wires together the Knowledge
//! Grounding Service, Formal Verifier, Arbiter, and Bias Detector into a
//! single, coherent anti‑hallucination pipeline.

use ahf_arbiter::Arbiter;
use ahf_bias_detector::BiasDetector;

    FormalVerificationResult, GroundingEvidence, GroundingScore, OutputResult, Result,

use ahf_knowledge_grounding::KnowledgeGroundingService;
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::RwLock;
use tracing::{info, warn};

// ---------------------------------------------------------------------------
//  Gateway Configuration
// ---------------------------------------------------------------------------

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct GatewayConfig {
    /// Default schema name to use for formal verification.
    pub default_schema: String,
    /// Whether to enable the AI advisor for enhanced NL parsing (feature‑gated).
    pub ai_enhanced_parsing: bool,
    /// Maximum number of concurrent pipeline executions.
    pub max_concurrency: usize,
    /// Session TTL in seconds.
    pub session_ttl_secs: u64,
}

impl Default for GatewayConfig {
    fn default() -> Self {
        Self {
            default_schema: "text".into(),
            ai_enhanced_parsing: false,
            max_concurrency: 100,
            session_ttl_secs: 3600,
        }
    }
}

// ---------------------------------------------------------------------------
//  Pipeline Statistics
// ---------------------------------------------------------------------------

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct PipelineStats {
    pub total_processed: u64,
    pub accepted: u64,
    pub rejected: u64,
    pub escalated: u64,
    pub avg_latency_us: u64,
}

// ---------------------------------------------------------------------------
//  AHF Gateway
// ---------------------------------------------------------------------------

pub struct AhfGateway {
    grounding: Arc<KnowledgeGroundingService>,
    verifier: Arc<FormalVerifier>,
    arbiter: Arc<Arbiter>,
    bias: Arc<BiasDetector>,
    config: Arc<RwLock<GatewayConfig>>,
    stats: Arc<RwLock<PipelineStats>>,
}

impl AhfGateway {
    /// Create a new gateway with the provided components.
    pub fn new(
        grounding: Arc<KnowledgeGroundingService>,
        verifier: Arc<FormalVerifier>,
        arbiter: Arc<Arbiter>,
        bias: Arc<BiasDetector>,
    ) -> Self {
        Self {
            grounding,
            verifier,
            arbiter,
            bias,
            config: Arc::new(RwLock::new(GatewayConfig::default())),
            stats: Arc::new(RwLock::new(PipelineStats::default())),
        }
    }

    /// Process a raw AI output through the complete anti‑hallucination pipeline.
    ///
    /// # Arguments
    /// * `raw_output` – the text produced by the AI model.
    /// * `schema_name` – the name of the formal verification schema to apply.
    /// * `model_confidence` – the model's own confidence (0.0 – 1.0).
    /// * `session_id` – optional session identifier for consistency tracking.
    ///
    /// # Returns
    /// An `OutputResult` indicating whether the output was accepted, and
    /// the final text (either the original with safety envelope applied,
    /// or a fallback response).
    pub async fn process_output(
        &self,
        raw_output: &str,
        schema_name: &str,
        model_confidence: f32,
        session_id: Option<&str>,
    ) -> Result<OutputResult> {
        let start = std::time::Instant::now();

        // 1. Check for empty output
        if raw_output.is_empty() {
            return Ok(OutputResult::reject(
                "Empty output received",
                "I cannot process an empty response.",
            ));
        }

        // 2. Extract factual claims from the output
        let claims = extract_claims(raw_output);

        // 3. Ground claims against the knowledge base
        let grounding_scores = self.grounding.ground_claims(&claims).await;
        let grounding_score = if claims.is_empty() {
            1.0 // No claims to verify → assume safe
        } else {
            let found = grounding_scores
                .iter()
                .filter(|s| matches!(s, GroundingScore::Found(_)))
                .count();
            found as f64 / claims.len() as f64
        };

        // Check for contradictions
        let has_contradiction = grounding_scores
            .iter()
            .any(|s| matches!(s, GroundingScore::Contradicted(_)));

        // If any claim is contradicted, grounding score is effectively 0
        let effective_grounding = if has_contradiction { 0.0 } else { grounding_score };

        // 4. Formal verification
        let verification = self
            .verifier
            .validate(raw_output, schema_name, session_id)
            .await
            .unwrap_or_else(|e| {
                warn!("Formal verification error: {}", e);
                FormalVerificationResult::Invalid(vec![e.to_string()])
            });

        // 5. Bias detection
        let bias_report = self.bias.detect(raw_output).await;

        // 6. Arbiter decision
        let decision = self
            .arbiter
            .decide(
                effective_grounding,
                &verification,
                bias_report.bias_score,
                model_confidence,
            )
            .await;

        // 7. Produce final output
        let final_output = if decision.accepted {
            apply_safety_envelope(raw_output, &decision.safety_envelope)
        } else {
            decision.fallback.clone()
        };

        // 8. Update statistics
        {
            let mut stats = self.stats.write().await;
            stats.total_processed += 1;
            if decision.accepted {
                stats.accepted += 1;
            } else {
                stats.rejected += 1;
            }
            let elapsed_us = start.elapsed().as_micros() as u64;
            stats.avg_latency_us =
                (stats.avg_latency_us * (stats.total_processed - 1) + elapsed_us)
                    / stats.total_processed;
        }

        // 9. Log the decision for audit
        info!(
            accepted = decision.accepted,
            grounding_score = effective_grounding,
            bias_score = bias_report.bias_score,
            confidence = model_confidence,
            reason = %decision.reason,
            "AHF pipeline decision"
        );

        Ok(OutputResult {
            accepted: decision.accepted,
            final_output,
            grounding_score: effective_grounding,
            verification_result: verification,
            bias_detected: bias_report.bias_score > 0.5,
            confidence: model_confidence,
            reason: decision.reason,
            fallback_used: !decision.accepted,
        })
    }

    /// Process output with the default schema.
    pub async fn process_output_default(
        &self,
        raw_output: &str,
        model_confidence: f32,
    ) -> Result<OutputResult> {
        let config = self.config.read().await;
        self.process_output(raw_output, &config.default_schema, model_confidence, None)
            .await
    }

    /// Update gateway configuration at runtime.
    pub async fn update_config(&self, new_config: GatewayConfig) {
        let mut config = self.config.write().await;
        *config = new_config;
    }

    /// Get pipeline statistics.
    pub async fn stats(&self) -> PipelineStats {
        self.stats.read().await.clone()
    }
}

        let mut obligations = self.obligations.write().await;
        if let Some(obl) = obligations.get_mut(id) {
            // In a real implementation, this would:
            // 1. Load the proof artefact from CAS by hash.
            // 2. Run the Axiom proof checker.
            // 3. Return the result.
            if obl.proof_hash.is_some() {

                        .duration_since(std::time::UNIX_EPOCH)
                        .unwrap()
                        .as_secs(),
                );
                Ok(true)

        let obligations = self.obligations.read().await;
        let total = obligations.len();
        let mut verified = 0;
        let mut failed = 0;
        let mut pending = 0;
        let mut results = Vec::new();

        for obl in obligations.values() {
            if obl.verified {
                verified += 1;
            } else if obl.proof_hash.is_some() {
                // Has a proof but not yet verified – try now
                // In production, this would run the checker.
                pending += 1;
            } else {
                failed += 1;
            }
            results.push(obl.clone());
        }

        VerificationReport {
            total,
            verified,
            failed,
            pending,
            obligations: results,
            healthy: failed == 0,
        }
    }

        let obligations = self.obligations.read().await;
        obligations.get(id).cloned()
    }

    pub async fn is_healthy(&self) -> bool {
        let report = self.verify_all().await;

    }
