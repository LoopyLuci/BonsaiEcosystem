### Using Hot-Reload Engine

```typescript
import { useOmnisystemTheme } from './OMNISYSTEM_HOT_RELOAD_ENGINE'

function MyComponent() {
  const { activateTheme, updateProperty, getAllThemes } = useOmnisystemTheme()

  return (
    <>
      <button onClick={() => activateTheme('dark_pure')}>
        Switch to Dark
      </button>
      <button onClick={() => updateProperty('primaryColor', '#FF006E')}>
        Change Color
      </button>
    </>
  )
}
```
