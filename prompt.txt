```rust
use pest::Parser;
use pest_derive::Parser;
use serde::{Serialize, Deserialize};

#[derive(Parser)]
#[grammar = "sylva.pest"]
pub struct SylvaParser;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum Expr {
    IntLiteral(i64),
    FloatLiteral(f64),
    StringLiteral(String),
    BoolLiteral(bool),
    Nil,
    Identifier(String),
    BinaryOp {
        op: BinOp,
        left: Box<Expr>,
        right: Box<Expr>,
    },
    UnaryOp {
        op: UnaryOp,
        expr: Box<Expr>,
    },
    Call {
        func: Box<Expr>,
        args: Vec<Expr>,
    },
    If {
        condition: Box<Expr>,
        then_branch: Box<Expr>,
        else_branch: Option<Box<Expr>>,
    },
    While {
        condition: Box<Expr>,
        body: Box<Expr>,
    },
    For {
        var: String,
        iterable: Box<Expr>,
        body: Box<Expr>,
    },
    FnDef {
        name: String,
        params: Vec<String>,
        body: Box<Expr>,
    },
    Return(Box<Expr>),
    Assign {
        name: String,
        value: Box<Expr>,
    },
    Block(Vec<Expr>),
    TableDef {
        entries: Vec<(Expr, Expr)>,
    },
    ListDef {
        items: Vec<Expr>,
    },
    Index {
        target: Box<Expr>,
        index: Box<Expr>,
    },
    Dot {
        target: Box<Expr>,
        field: String,
    },
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum BinOp {
    Add, Sub, Mul, Div, Mod, Pow,
    Eq, Neq, Lt, Gt, Le, Ge,
    And, Or,
    Concat,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum UnaryOp {
    Neg, Not, Len,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Program {
    pub statements: Vec<Expr>,
}

pub fn parse(source: &str) -> anyhow::Result<Program> {
    let pairs = SylvaParser::parse(Rule::program, source)?
        .next()
        .ok_or_else(|| anyhow::anyhow!("Empty program"))?;
    
    let mut statements = Vec::new();
    for pair in pairs.into_inner() {
        match pair.as_rule() {
            Rule::statement => {
                statements.push(parse_expr(pair.into_inner().next().unwrap())?);
            }
            Rule::EOI => break,
            _ => {}
        }
    }
    
    Ok(Program { statements })
}

fn parse_expr(pair: pest::iterators::Pair<Rule>) -> anyhow::Result<Expr> {
    match pair.as_rule() {
        Rule::integer => {
            let s = pair.as_str();
            if s.starts_with("0x") {
                Ok(Expr::IntLiteral(i64::from_str_radix(&s[2..], 16)?))
            } else {
                Ok(Expr::IntLiteral(s.parse()?))
            }
        }
        Rule::float => Ok(Expr::FloatLiteral(pair.as_str().parse()?)),
        Rule::string => {
            let s = pair.as_str();
            Ok(Expr::StringLiteral(s[1..s.len()-1].to_string()))
        }
        Rule::bool_true => Ok(Expr::BoolLiteral(true)),
        Rule::bool_false => Ok(Expr::BoolLiteral(false)),
        Rule::nil => Ok(Expr::Nil),
        Rule::identifier => Ok(Expr::Identifier(pair.as_str().to_string())),
        Rule::call => {
            let mut inner = pair.into_inner();
            let func = Box::new(parse_expr(inner.next().unwrap())?);
            let mut args = Vec::new();
            for arg in inner {
                args.push(parse_expr(arg)?);
            }
            Ok(Expr::Call { func, args })
        }
        Rule::fn_def => {
            let mut inner = pair.into_inner();
            let name = inner.next().unwrap().as_str().to_string();
            let params_str = inner.next().unwrap().as_str();
            let params: Vec<String> = params_str[1..params_str.len()-1]
                .split(',')
                .map(|s| s.trim().to_string())
                .filter(|s| !s.is_empty())
                .collect();
            let body = Box::new(parse_expr(inner.next().unwrap())?);
            Ok(Expr::FnDef { name, params, body })
        }
        Rule::if_expr => {
            let mut inner = pair.into_inner();
            let condition = Box::new(parse_expr(inner.next().unwrap())?);
            let then_branch = Box::new(parse_expr(inner.next().unwrap())?);
            let else_branch = inner.next().map(|e| Box::new(parse_expr(e).unwrap()));
            Ok(Expr::If { condition, then_branch, else_branch })
        }
        Rule::while_expr => {
            let mut inner = pair.into_inner();
            let condition = Box::new(parse_expr(inner.next().unwrap())?);
            let body = Box::new(parse_expr(inner.next().unwrap())?);
            Ok(Expr::While { condition, body })
        }
        Rule::return_expr => {
            let mut inner = pair.into_inner();
            Ok(Expr::Return(Box::new(parse_expr(inner.next().unwrap())?)))
        }
        Rule::assign => {
            let mut inner = pair.into_inner();
            let name = inner.next().unwrap().as_str().to_string();
            let value = Box::new(parse_expr(inner.next().unwrap())?);
            Ok(Expr::Assign { name, value })
        }
        Rule::block => {
            let mut statements = Vec::new();
            for stmt in pair.into_inner() {
                statements.push(parse_expr(stmt)?);
            }
            Ok(Expr::Block(statements))
        }
        _ => {
            // Binary operations handled by precedence climbing
            parse_binary(pair)
        }
    }
}

fn parse_binary(pair: pest::iterators::Pair<Rule>) -> anyhow::Result<Expr> {
    // Simplified: real parser uses Pratt parsing
    Ok(Expr::IntLiteral(0))
}
```

```rust
use crate::parser::{Expr, BinOp, UnaryOp};

/// Bytecode instructions for the Sylva VM.
#[derive(Debug, Clone)]
pub enum OpCode {
    // Stack
    PushInt(i64),
    PushFloat(f64),
    PushString(String),
    PushBool(bool),
    PushNil,
    Pop,
    Dup,
    
    // Variables
    LoadVar(String),
    StoreVar(String),
    LoadGlobal(String),
    StoreGlobal(String),
    
    // Arithmetic
    Add, Sub, Mul, Div, Mod, Pow,
    Neg,
    
    // Comparison
    Eq, Neq, Lt, Gt, Le, Ge,
    
    // Logic
    And, Or, Not,
    
    // Control Flow
    Jump(usize),
    JumpIfFalse(usize),
    JumpIfTrue(usize),
    
    // Functions
    Call(u8),
    Return,
    MakeFunc {
        name: String,
        param_count: u8,
        body_offset: usize,
    },
    
    // Data structures
    MakeList(u32),
    MakeTable(u32),
    IndexGet,
    IndexSet,
    DotGet(String),
    
    // Built-in
    Print,
    Len,
    TypeOf,
    
    // Concurrency
    Spawn,
    Send,
    Receive,
    
    Halt,
}

pub fn compile(program: &crate::parser::Program) -> Vec<OpCode> {
    let mut code = Vec::new();
    for stmt in &program.statements {
        compile_expr(stmt, &mut code);
    }
    code.push(OpCode::Halt);
    code
}

fn compile_expr(expr: &Expr, code: &mut Vec<OpCode>) {
    match expr {
        Expr::IntLiteral(v) => code.push(OpCode::PushInt(*v)),
        Expr::FloatLiteral(v) => code.push(OpCode::PushFloat(*v)),
        Expr::StringLiteral(v) => code.push(OpCode::PushString(v.clone())),
        Expr::BoolLiteral(v) => code.push(OpCode::PushBool(*v)),
        Expr::Nil => code.push(OpCode::PushNil),
        Expr::Identifier(name) => code.push(OpCode::LoadVar(name.clone())),
        Expr::BinaryOp { op, left, right } => {
            compile_expr(left, code);
            compile_expr(right, code);
            match op {
                BinOp::Add => code.push(OpCode::Add),
                BinOp::Sub => code.push(OpCode::Sub),
                BinOp::Mul => code.push(OpCode::Mul),
                BinOp::Div => code.push(OpCode::Div),
                BinOp::Mod => code.push(OpCode::Mod),
                BinOp::Pow => code.push(OpCode::Pow),
                BinOp::Eq => code.push(OpCode::Eq),
                BinOp::Neq => code.push(OpCode::Neq),
                BinOp::Lt => code.push(OpCode::Lt),
                BinOp::Gt => code.push(OpCode::Gt),
                BinOp::Le => code.push(OpCode::Le),
                BinOp::Ge => code.push(OpCode::Ge),
                BinOp::And => code.push(OpCode::And),
                BinOp::Or => code.push(OpCode::Or),
                BinOp::Concat => {
                    // Concat is handled by the VM as string operations
                    code.push(OpCode::Call(2)); // Two args: left and right
                }
            }
        }
        Expr::UnaryOp { op, expr: inner } => {
            compile_expr(inner, code);
            match op {
                UnaryOp::Neg => code.push(OpCode::Neg),
                UnaryOp::Not => code.push(OpCode::Not),
                UnaryOp::Len => code.push(OpCode::Len),
            }
        }
        Expr::Call { func, args } => {
            for arg in args.iter().rev() {
                compile_expr(arg, code);
            }
            compile_expr(func, code);
            code.push(OpCode::Call(args.len() as u8));
        }
        Expr::Assign { name, value } => {
            compile_expr(value, code);
            code.push(OpCode::StoreVar(name.clone()));
        }
        Expr::Block(stmts) => {
            for stmt in stmts {
                compile_expr(stmt, code);
            }
        }
        Expr::Return(v) => {
            compile_expr(v, code);
            code.push(OpCode::Return);
        }
        Expr::If { condition, then_branch, else_branch } => {
            compile_expr(condition, code);
            let jump_idx = code.len();
            code.push(OpCode::JumpIfFalse(0)); // placeholder
            compile_expr(then_branch, code);
            if let Some(else_b) = else_branch {
                let skip_idx = code.len();
                code.push(OpCode::Jump(0)); // placeholder
                code[jump_idx] = OpCode::JumpIfFalse(code.len());
                compile_expr(else_b, code);
                code[skip_idx] = OpCode::Jump(code.len());
            } else {
                code[jump_idx] = OpCode::JumpIfFalse(code.len());
            }
        }
        Expr::While { condition, body } => {
            let loop_start = code.len();
            compile_expr(condition, code);
            let jump_idx = code.len();
            code.push(OpCode::JumpIfFalse(0));
            compile_expr(body, code);
            code.push(OpCode::Jump(loop_start));
            code[jump_idx] = OpCode::JumpIfFalse(code.len());
        }
        Expr::ListDef { items } => {
            for item in items.iter().rev() {
                compile_expr(item, code);
            }
            code.push(OpCode::MakeList(items.len() as u32));
        }
        Expr::TableDef { entries } => {
            for (k, v) in entries.iter().rev() {
                compile_expr(k, code);
                compile_expr(v, code);
            }
            code.push(OpCode::MakeTable((entries.len() * 2) as u32));
        }
        Expr::FnDef { name, params, body } => {
            let body_offset = code.len();
            // Skip over the function body and jump to it later
            code.push(OpCode::MakeFunc {
                name: name.clone(),
                param_count: params.len() as u8,
                body_offset,
            });
            code.push(OpCode::Jump(0)); // placeholder
        }
        _ => {}
    }
}
```

### `crates/omnisystem-sylva/src/vm.rs`

```rust
use crate::compiler::OpCode;
use std::collections::HashMap;

#[derive(Debug, Clone)]
pub enum Value {
    Int(i64),
    Float(f64),
    String(String),
    Bool(bool),
    Nil,
    Function {
        name: String,
        param_count: u8,
        body_offset: usize,
    },
    List(Vec<Value>),
    Table(HashMap<String, Value>),
}

impl Value {
    pub fn as_int(&self) -> Option<i64> {
        match self { Value::Int(v) => Some(*v), _ => None }
    }
    pub fn as_float(&self) -> Option<f64> {
        match self { Value::Float(v) => Some(*v), _ => None }
    }
    pub fn as_string(&self) -> Option<&str> {
        match self { Value::String(v) => Some(v), _ => None }
    }
    pub fn as_bool(&self) -> Option<bool> {
        match self { Value::Bool(v) => Some(*v), _ => None }
    }
}

pub struct Vm {
    code: Vec<OpCode>,
    stack: Vec<Value>,
    locals: Vec<HashMap<String, Value>>,
    globals: HashMap<String, Value>,
    ip: usize,
    call_stack: Vec<usize>,
}

impl Vm {
    pub fn new(code: Vec<OpCode>) -> Self {
        Self {
            code,
            stack: Vec::new(),
            locals: vec![HashMap::new()],
            globals: HashMap::new(),
            ip: 0,
            call_stack: Vec::new(),
        }
    }
    
    pub fn run(&mut self) -> anyhow::Result<Value> {
        loop {
            if self.ip >= self.code.len() {
                break;
            }
            let op = self.code[self.ip].clone();
            self.ip += 1;
            
            match op {
                OpCode::Halt => break,
                OpCode::PushInt(v) => self.stack.push(Value::Int(v)),
                OpCode::PushFloat(v) => self.stack.push(Value::Float(v)),
                OpCode::PushString(v) => self.stack.push(Value::String(v)),
                OpCode::PushBool(v) => self.stack.push(Value::Bool(v)),
                OpCode::PushNil => self.stack.push(Value::Nil),
                OpCode::Pop => { self.stack.pop(); }
                OpCode::Dup => {
                    let v = self.stack.last().cloned().unwrap_or(Value::Nil);
                    self.stack.push(v);
                }
                OpCode::LoadVar(name) => {
                    let val = self.locals.last()
                        .and_then(|m| m.get(&name))
                        .cloned()
                        .or_else(|| self.globals.get(&name).cloned())
                        .unwrap_or(Value::Nil);
                    self.stack.push(val);
                }
                OpCode::StoreVar(name) => {
                    let val = self.stack.pop().unwrap_or(Value::Nil);
                    if let Some(scope) = self.locals.last_mut() {
                        scope.insert(name, val);
                    }
                }
                OpCode::Add => self.binary_op(|a, b| match (a, b) {
                    (Value::Int(x), Value::Int(y)) => Value::Int(x + y),
                    (Value::Float(x), Value::Float(y)) => Value::Float(x + y),
                    (Value::String(x), Value::String(y)) => Value::String(format!("{}{}", x, y)),
                    _ => Value::Nil,
                }),
                OpCode::Sub => self.binary_op(|a, b| match (a, b) {
                    (Value::Int(x), Value::Int(y)) => Value::Int(x - y),
                    (Value::Float(x), Value::Float(y)) => Value::Float(x - y),
                    _ => Value::Nil,
                }),
                OpCode::Mul => self.binary_op(|a, b| match (a, b) {
                    (Value::Int(x), Value::Int(y)) => Value::Int(x * y),
                    (Value::Float(x), Value::Float(y)) => Value::Float(x * y),
                    _ => Value::Nil,
                }),
                OpCode::Div => self.binary_op(|a, b| match (a, b) {
                    (Value::Int(x), Value::Int(y)) if y != 0 => Value::Int(x / y),
                    (Value::Float(x), Value::Float(y)) if y != 0.0 => Value::Float(x / y),
                    _ => Value::Nil,
                }),
                OpCode::Eq => self.binary_op(|a, b| {
                    Value::Bool(std::mem::discriminant(&a) == std::mem::discriminant(&b))
                }),
                OpCode::Print => {
                    let val = self.stack.pop().unwrap_or(Value::Nil);
                    match &val {
                        Value::String(s) => println!("{}", s),
                        Value::Int(v) => println!("{}", v),
                        Value::Float(v) => println!("{}", v),
                        Value::Bool(v) => println!("{}", v),
                        Value::Nil => println!("nil"),
                        _ => println!("{:?}", val),
                    }
                    self.stack.push(val);
                }
                OpCode::Call(nargs) => {
                    let func = self.stack[self.stack.len() - 1 - nargs as usize].clone();
                    let args: Vec<Value> = self.stack.drain(self.stack.len() - nargs as usize..).collect();
                    self.stack.pop(); // Remove function
                    
                    match func {
                        Value::Function { param_count, body_offset, .. } => {
                            self.call_stack.push(self.ip);
                            self.locals.push(HashMap::new());
                            self.ip = body_offset;
                        }
                        _ => {
                            // Built-in function
                            self.stack.push(Value::Nil);
                        }
                    }
                }
                OpCode::Return => {
                    if let Some(ret_addr) = self.call_stack.pop() {
                        self.locals.pop();
                        self.ip = ret_addr;
                    }
                }
                OpCode::Jump(addr) => self.ip = addr,
                OpCode::JumpIfFalse(addr) => {
                    let val = self.stack.pop().unwrap_or(Value::Nil);
                    if !val.as_bool().unwrap_or(true) {
                        self.ip = addr;
                    }
                }
                OpCode::MakeList(n) => {
                    let items: Vec<Value> = self.stack.drain(self.stack.len() - n as usize..).collect();
                    self.stack.push(Value::List(items));
                }
                OpCode::MakeTable(n) => {
                    let pairs: Vec<Value> = self.stack.drain(self.stack.len() - n as usize..).collect();
                    let mut table = HashMap::new();
                    for chunk in pairs.chunks(2) {
                        if chunk.len() == 2 {
                            if let Some(key) = chunk[0].as_string() {
                                table.insert(key.to_string(), chunk[1].clone());
                            }
                        }
                    }
                    self.stack.push(Value::Table(table));
                }
                _ => {}
            }
        }
        
        Ok(self.stack.pop().unwrap_or(Value::Nil))
    }
    
    fn binary_op<F>(&mut self, f: F)
    where F: FnOnce(Value, Value) -> Value
    {
        let b = self.stack.pop().unwrap_or(Value::Nil);
        let a = self.stack.pop().unwrap_or(Value::Nil);
        self.stack.push(f(a, b));
    }
}
```

### `crates/omnisystem-sylva/src/jit.rs`

```rust
use cranelift::prelude::*;
use cranelift_module::{Module, FuncId};
use cranelift_jit::{JITBuilder, JITModule};
use crate::compiler::OpCode;
use std::collections::HashMap;

pub struct SylvaJit {
    module: JITModule,
    func_ids: HashMap<String, FuncId>,
    builder_ctx: FunctionBuilderContext,
}

impl SylvaJit {
    pub fn new() -> anyhow::Result<Self> {
        let mut flag_builder = settings::builder();
        flag_builder.set("use_colocated_libcalls", "false")?;
        flag_builder.set("is_pic", "true")?;
        let isa_builder = cranelift_native::builder()?;
        let isa = isa_builder.finish(settings::Flags::new(flag_builder))?;
        let builder = JITBuilder::with_isa(isa, cranelift_module::default_libcall_names());
        let module = JITModule::new(builder);
        Ok(Self {
            module,
            func_ids: HashMap::new(),
            builder_ctx: FunctionBuilderContext::new(),
        })
    }
    
    /// Compile a bytecode function into native machine code.
    pub fn compile_function(&mut self, name: &str, code: &[OpCode]) -> anyhow::Result<*const u8> {
        let mut sig = self.module.make_signature();
        sig.params.push(AbiParam::new(types::I64)); // args pointer
        sig.returns.push(AbiParam::new(types::I64)); // return value
        
        let func_id = self.module.declare_function(name, Linkage::Local, &sig)?;
        let mut func_ctx = self.module.make_context();
        func_ctx.func.signature = sig.clone();
        func_ctx.func.name = ExternalName::user(0, func_id.as_u32());
        
        let mut builder = FunctionBuilder::new(&mut func_ctx.func, &mut self.builder_ctx);
        let entry_block = builder.create_block();
        builder.switch_to_block(entry_block);
        builder.seal_block(entry_block);
        
        // Simplified: return a constant for now
        let result = builder.ins().iconst(types::I64, 42);
        builder.ins().return_(&[result]);
        builder.finalize();
        
        self.module.define_function(func_id, &mut func_ctx)?;
        self.module.clear_context(&mut func_ctx);
        self.module.finalize_definitions()?;
        
        let code_ptr = self.module.get_finalized_function(func_id);
        self.func_ids.insert(name.to_string(), func_id);
        Ok(code_ptr)
    }
    
    /// Execute a JIT-compiled function.
    pub fn call(&self, name: &str, args: &[i64]) -> anyhow::Result<i64> {
        if let Some(func_id) = self.func_ids.get(name) {
```

```rust
use crate::ast::*;
use std::collections::{HashMap, HashSet};
use thiserror::Error;

#[derive(Error, Debug)]
pub enum TypeError {
    #[error("Type mismatch: expected {expected:?}, got {actual:?}")]
    TypeMismatch { expected: Type, actual: Type },
    #[error("Undefined variable: {0}")]
    UndefinedVariable(String),
    #[error("Undefined type: {0}")]
    UndefinedType(String),
    #[error("Cannot infer type")]
    InferenceFailure,
    #[error("Effect violation: {0}")]
    EffectViolation(String),
    #[error("Borrow check violation: {0}")]
    BorrowViolation(String),
}

pub struct TypeChecker {
    types: HashMap<String, TypeInfo>,
    variables: Vec<HashMap<String, Type>>,
    functions: HashMap<String, Function>,
    structs: HashMap<String, StructDef>,
    enums: HashMap<String, EnumDef>,
    traits: HashMap<String, TraitDef>,
    current_function: Option<String>,
    errors: Vec<TypeError>,
}

#[derive(Debug, Clone)]
struct TypeInfo {
    kind: TypeKind,
    generics: Vec<String>,
}

#[derive(Debug, Clone)]
enum TypeKind {
    Struct(StructDef),
    Enum(EnumDef),
    Trait(TraitDef),
    Alias(Type),
}

impl TypeChecker {
    pub fn new() -> Self {
        Self {
            types: HashMap::new(),
            variables: vec![HashMap::new()],
            functions: HashMap::new(),
            structs: HashMap::new(),
            enums: HashMap::new(),
            traits: HashMap::new(),
            current_function: None,
            errors: Vec::new(),
        }
    }
    
    pub fn check(&mut self, program: &Program) -> Result<(), Vec<TypeError>> {
        // First pass: collect all type definitions
        for item in &program.items {
            match item {
                Item::Struct(s) => {
                    self.structs.insert(s.name.clone(), s.clone());
                    self.types.insert(s.name.clone(), TypeInfo {
                        kind: TypeKind::Struct(s.clone()),
                        generics: s.generics.clone(),
                    });
                }
                Item::Enum(e) => {
                    self.enums.insert(e.name.clone(), e.clone());
                    self.types.insert(e.name.clone(), TypeInfo {
                        kind: TypeKind::Enum(e.clone()),
                        generics: e.generics.clone(),
                    });
                }
                Item::Trait(t) => {
                    self.traits.insert(t.name.clone(), t.clone());
                    self.types.insert(t.name.clone(), TypeInfo {
                        kind: TypeKind::Trait(t.clone()),
                        generics: t.generics.clone(),
                    });
                }
                _ => {}
            }
        }
        
        // Second pass: register function signatures
        for item in &program.items {
            if let Item::Function(f) = item {
                self.functions.insert(f.name.clone(), f.clone());
            }
        }
        
        // Third pass: type check function bodies
        for item in &program.items {
            if let Item::Function(f) = item {
                self.current_function = Some(f.name.clone());
                self.push_scope();
                for param in &f.params {
                    self.variables.last_mut().unwrap().insert(param.name.clone(), param.ty.clone());
                }
                let body_type = self.check_block(&f.body);
                if let Some(ret_type) = body_type {
                    self.unify(&ret_type, &f.return_type);
                }
                self.pop_scope();
            }
        }
        
        if self.errors.is_empty() {
            Ok(())
        } else {
            Err(self.errors.clone())
        }
    }
    
    fn check_block(&mut self, block: &Block) -> Option<Type> {
        let mut last_type = None;
        for stmt in &block.stmts {
            last_type = self.check_stmt(stmt);
        }
        last_type
    }
    
    fn check_stmt(&mut self, stmt: &Stmt) -> Option<Type> {
        match stmt {
            Stmt::Let { name, ty, value } => {
                let value_type = self.check_expr(value);
                if let Some(expected) = ty {
                    self.unify(expected, &value_type);
                }
                self.variables.last_mut().unwrap().insert(name.clone(), ty.clone().unwrap_or(value_type));
                None
            }
            Stmt::Expr(e) => { Some(self.check_expr(e)) }
            Stmt::Return(e) => {
                if let Some(val) = e {
                    Some(self.check_expr(val))
                } else {
                    Some(Type::Never)
                }
            }
            Stmt::If { condition, then_block, else_block } => {
                let cond_type = self.check_expr(condition);
                self.unify(&cond_type, &Type::Named("bool".into()));
                let then_type = self.check_block(then_block);
                if let Some(else_b) = else_block {
                    let else_type = self.check_block(else_b);
                    // Both branches must have the same type
                    if let (Some(t1), Some(t2)) = (&then_type, &else_type) {
                        self.unify(t1, t2);
                    }
                    then_type
                } else {
                    None
                }
            }
            _ => None,
        }
    }
    
    fn check_expr(&mut self, expr: &Expr) -> Type {
        match expr {
            Expr::IntLiteral(_) => Type::Named("i64".into()),
            Expr::FloatLiteral(_) => Type::Named("f64".into()),
            Expr::StringLiteral(_) => Type::Named("String".into()),
            Expr::BoolLiteral(_) => Type::Named("bool".into()),
            Expr::CharLiteral(_) => Type::Named("char".into()),
            Expr::Identifier(name) => {
                for scope in self.variables.iter().rev() {
                    if let Some(ty) = scope.get(name) {
                        return ty.clone();
                    }
                }
                self.errors.push(TypeError::UndefinedVariable(name.clone()));
                Type::Infer
            }
            Expr::Binary { op, left, right } => {
                let lt = self.check_expr(left);
                let rt = self.check_expr(right);
                self.unify(&lt, &rt);
                lt
            }
            Expr::Call { func, args } => {
                let func_type = self.check_expr(func);
                for arg in args {
                    self.check_expr(arg);
                }
                Type::Infer
            }
            _ => Type::Infer,
        }
    }
    
    fn unify(&mut self, expected: &Type, actual: &Type) {
        if matches!(expected, Type::Infer) || matches!(actual, Type::Infer) {
            return;
        }
        if std::mem::discriminant(expected) != std::mem::discriminant(actual) {
            self.errors.push(TypeError::TypeMismatch {
                expected: expected.clone(),
                actual: actual.clone(),
            });
        }
    }
    
    fn push_scope(&mut self) { self.variables.push(HashMap::new()); }
    fn pop_scope(&mut self) { self.variables.pop(); }
}
```

```rust
use crate::ast::*;
use bonsai_lair::*;
use std::collections::HashMap;

pub fn lower_program(program: &Program) -> LairModule {
    let mut module = LairModule {
        name: "titan_module".into(),
        language: "Titan".into(),
        version: "0.1.0".into(),
        functions: Vec::new(),
        types: Vec::new(),
        imports: Vec::new(),
        exports: Vec::new(),
    };
    
    for item in &program.items {
        if let Item::Function(func) = item {
            let lair_func = lower_function(func);
            module.functions.push(lair_func);
        }
    }
    
    module
}

fn lower_function(func: &Function) -> LairFunction {
    let mut ctx = LowerCtx::new();
    for param in &func.params {
        ctx.add_local(&param.name);
    }
    
    let body = lower_block(&func.body, &mut ctx);
    let signature = FunctionSignature {
        params: func.params.iter().map(|p| (p.name.clone(), lower_type(&p.ty))).collect(),
        return_type: lower_type(&func.return_type),
    };
    
    LairFunction {
        name: func.name.clone(),
        signature,
        body,
        effects: func.effects.iter().map(|e| match e {
            Effect::IO => bonsai_lair::Effect::Io,
            Effect::Alloc => bonsai_lair::Effect::Alloc,
            Effect::Async => bonsai_lair::Effect::Async,
            Effect::Panic => bonsai_lair::Effect::Panic,
            Effect::Pure => bonsai_lair::Effect::Pure,
        }).collect(),
        metadata: FunctionMetadata {
            version_hash: [0u8; 32],
            safe_points: vec![],
            migration: None,
            source_language: "Titan".into(),
            source_file: String::new(),
            source_line: 0,
        },
    }
}

fn lower_type(ty: &Type) -> BuirType {
    match ty {
        Type::Named(s) if s == "i64" => BuirType::I64,
        Type::Named(s) if s == "i32" => BuirType::I32,
        Type::Named(s) if s == "f64" => BuirType::F64,
        Type::Named(s) if s == "bool" => BuirType::Bool,
        Type::Named(s) if s == "String" => BuirType::String,
        Type::Named(_) => BuirType::Foreign(ty_to_string(ty)),
        Type::Ptr(t) => BuirType::Pointer(Box::new(lower_type(t))),
        Type::Ref(t, _) => BuirType::Reference(Box::new(lower_type(t))),
        _ => BuirType::Void,
    }
}

fn ty_to_string(ty: &Type) -> String {
    match ty {
        Type::Named(s) => s.clone(),
        _ => format!("{:?}", ty),
    }
}

struct LowerCtx {
    locals: Vec<String>,
    temp_counter: u32,
}

impl LowerCtx {
    fn new() -> Self { Self { locals: Vec::new(), temp_counter: 0 } }
    fn add_local(&mut self, name: &str) { self.locals.push(name.to_string()); }
    fn new_temp(&mut self) -> String {
        let t = format!("t{}", self.temp_counter);
        self.temp_counter += 1;
        t
    }
}

fn lower_block(block: &Block, ctx: &mut LowerCtx) -> Vec<LairNode> {
    let mut nodes = Vec::new();
    for stmt in &block.stmts {
        nodes.extend(lower_stmt(stmt, ctx));
    }
    nodes
}

fn lower_stmt(stmt: &Stmt, ctx: &mut LowerCtx) -> Vec<LairNode> {
    match stmt {
        Stmt::Return(Some(expr)) => {
            let val = lower_expr(expr, ctx);
            let mut nodes = val.nodes;
            nodes.push(LairNode::Return { value: Some(val.target) });
            nodes
        }
        Stmt::Let { name, value, .. } => {
            let val = lower_expr(value, ctx);
            ctx.add_local(name);
            let mut nodes = val.nodes;
            nodes.push(LairNode::Store {
                target: name.clone(),
                value: val.target,
            });
            nodes
        }
        _ => vec![LairNode::Return { value: None }],
    }
}

struct LoweredExpr {
    nodes: Vec<LairNode>,
    target: String,
}

fn lower_expr(expr: &Expr, ctx: &mut LowerCtx) -> LoweredExpr {
    match expr {
        Expr::IntLiteral(v) => {
            let t = ctx.new_temp();
            LoweredExpr {
                nodes: vec![],
                target: format!("{}", v),
            }
        }
        Expr::Binary { op, left, right } => {
            let l = lower_expr(left, ctx);
            let r = lower_expr(right, ctx);
            let t = ctx.new_temp();
            let mut nodes = l.nodes;
            nodes.extend(r.nodes);
            nodes.push(LairNode::Call {
                target: t.clone(),
                function: format!("builtin_{:?}", op).to_lowercase(),
                args: vec![l.target, r.target],
            });
            LoweredExpr { nodes, target: t }
        }
        _ => {
            let t = ctx.new_temp();
            LoweredExpr { nodes: vec![], target: t }
        }
    }
}
```

```rust
        self.builder.build_return(Some(&i64_type.const_int(42, false)));

    }
```

```rust
pub fn parse(source: &str) -> anyhow::Result<Program> {
    if source.trim().is_empty() {
        return Ok(Program { items: vec![] });
    }
    
    // Simple parsing: detect function definitions
    let mut items = Vec::new();
    for line in source.lines() {
        let line = line.trim();
        if line.starts_with("fn ") {
            let rest = &line[3..];
            let paren_idx = rest.find('(').unwrap_or(rest.len());
            let name = rest[..paren_idx].trim().to_string();
            
            items.push(Item::Function(Function {
                name,
                params: vec![],
                return_type: Type::Named("i64".into()),
                body: Block { stmts: vec![Stmt::Return(Some(Expr::IntLiteral(42)))] },
                effects: vec![Effect::Pure],
                generics: vec![],
                attributes: vec![],
            }));
        }
    }
    
    Ok(Program { items })
}
```

```rust
use bonsai_lair::*;
use sqlparser::ast::*;

pub fn statement_to_lair(stmt: &Statement) -> LairFunction {
    match stmt {
        Statement::Query(query) => query_to_lair(query),
        _ => LairFunction {
            name: "sql_statement".into(),
            signature: FunctionSignature { params: vec![], return_type: BuirType::Dynamic },
            body: vec![],
            effects: vec![],
            metadata: Default::default(),
        },
    }
}

fn query_to_lair(query: &Query) -> LairFunction {
    let mut body = Vec::new();
    if let SetExpr::Select(select) = &*query.body {
        let table = select.from.first()
            .and_then(|t| match &t.relation {
                TableFactor::Table { name, .. } => Some(name.to_string()),
                _ => None,
            })
            .unwrap_or_else(|| "unknown".to_string());
        
        let columns = select.projection.iter()
            .filter_map(|s| match s {
                SelectItem::UnnamedExpr(expr) => Some(format!("{:?}", expr)),
                SelectItem::ExprWithAlias { expr, alias } => Some(alias.value.clone()),
                _ => None,
            })
            .collect::<Vec<_>>()
            .join(", ");
        
        body.push(LairNode::Call {
            target: "result".into(),
            function: "sql_query".into(),
            args: vec![table, columns],
        });
    }
    
    LairFunction {
        name: "sql_query".into(),
        signature: FunctionSignature { params: vec![], return_type: BuirType::Dynamic },
        body,
        effects: vec![Effect::Io],
        metadata: Default::default(),
    }
}
```

```rust
        let (tx, rx) = mpsc::channel(256);
        Self {
            id: uuid::Uuid::new_v4().to_string(),
            state: def.initial_state.clone(),
            mailbox_rx: rx,
            mailbox_tx: tx,
            is_running: true,
            def,
        }
    }

        self.mailbox_tx.send(msg).await?;
        Ok(())
    }
    
    pub async fn run(&mut self) {
        while let Some(msg) = self.mailbox_rx.recv().await {
            match msg {
                ActorMessage::System(SystemMessage::Stop) => {
                    self.is_running = false;
                    break;
                }
                ActorMessage::System(SystemMessage::PoisonPill) => {
                    self.is_running = false;
                    break;
                }
                ActorMessage::User(data) => {
                    // Process user message
                    self.state = self.handle_message(data).unwrap_or(self.state.clone());
                }
                _ => {}
            }
        }
    }
    
    fn handle_message(&self, msg: serde_json::Value) -> Option<serde_json::Value> {
        // In production: call the handler function via BACE
        Some(msg)
    }
```

```rust
use crate::actor::{ActorInstance, ActorDef, SupervisionStrategy};
use std::collections::HashMap;
use tokio::sync::RwLock;
use std::sync::Arc;

pub struct Supervisor {
    pub name: String,
    pub strategy: SupervisionStrategy,
    pub children: Arc<RwLock<HashMap<String, ActorInstance>>>,
    pub max_restarts: u32,
    pub restart_window_secs: u64,
    pub restart_count: Arc<RwLock<HashMap<String, u32>>>,
}

impl Supervisor {
    pub fn new(name: &str, strategy: SupervisionStrategy) -> Self {
        Self {
            name: name.to_string(),
            strategy,
            children: Arc::new(RwLock::new(HashMap::new())),
            max_restarts: 3,
            restart_window_secs: 60,
            restart_count: Arc::new(RwLock::new(HashMap::new())),
        }
    }
    
    pub async fn spawn(&self, def: ActorDef) -> String {
        let actor = ActorInstance::new(def);
        let id = actor.id.clone();
        self.children.write().await.insert(id.clone(), actor);
        id
    }
    
    pub async fn stop(&self, child_id: &str) {
        if let Some(actor) = self.children.write().await.get_mut(child_id) {
            let _ = actor.send(crate::actor::ActorMessage::System(
                crate::actor::SystemMessage::Stop
            )).await;
        }
    }
    
    pub async fn handle_failure(&self, child_id: &str) -> anyhow::Result<()> {
        let mut counts = self.restart_count.write().await;
        let count = counts.entry(child_id.to_string()).or_insert(0);
        *count += 1;
        
        if *count > self.max_restarts {
            // Escalate: stop all children
            self.stop_all().await;
            anyhow::bail!("Max restarts exceeded for actor {}", child_id);
        }
        
        // Restart the actor
        if let Some(actor) = self.children.write().await.get(child_id) {
            let def = actor.def.clone();
            let new_actor = ActorInstance::new(def);
            self.children.write().await.insert(child_id.to_string(), new_actor);
        }
        
        Ok(())
    }
    
    pub async fn stop_all(&self) {
        for (id, _) in self.children.read().await.iter() {
            self.stop(id).await;
        }
    }
}
```

```rust
    }
```

```rust
            self.verified_proofs.insert(proof.name.clone(), verified);
        }
        valid
    }
```

```rust
pub mod profiler;
pub mod optimizer;
pub mod cache;
pub mod inline;

use bonsai_lair::LairFunction;

/// The JIT optimization pipeline.
pub struct JitOptimizer {
    profiler: profiler::HotspotProfiler,
    optimizer: optimizer::PassManager,
    cache: cache::CodeCache,
    inliner: inline::Inliner,
}

impl JitOptimizer {
    pub fn new() -> Self {
        Self {
            profiler: profiler::HotspotProfiler::new(),
            optimizer: optimizer::PassManager::default(),
            cache: cache::CodeCache::new(),
            inliner: inline::Inliner::new(),
        }
    }
    
    /// Optimize a LAIR function using profiling data.
    pub fn optimize(&mut self, func: &mut LairFunction) -> anyhow::Result<()> {
        let hash = blake3::hash(&serde_json::to_vec(&func.body).unwrap_or_default());
        
        // Check cache
        if let Some(cached) = self.cache.get(&hash) {
            func.body = cached;
            return Ok(());
        }
        
        // Profile: identify hot paths
        let hotspots = self.profiler.analyze(func);
        
        // Inline small functions
        self.inliner.inline_calls(func, &hotspots);
        
        // Run optimization passes
        self.optimizer.run(func);
        
        // Cache the result
        self.cache.put(hash, func.body.clone());
        
        Ok(())
    }
}
```

```rust
use bonsai_lair::{LairFunction, LairNode};
use std::collections::{HashMap, HashSet};

pub struct HotspotProfiler {
    call_counts: HashMap<String, u64>,
    loop_iterations: HashMap<usize, u64>,
}

impl HotspotProfiler {
    pub fn new() -> Self {
        Self {
            call_counts: HashMap::new(),
            loop_iterations: HashMap::new(),
        }
    }
    
    /// Analyze a function and return a list of "hot" node indices.
    pub fn analyze(&mut self, func: &LairFunction) -> HashSet<usize> {
        let mut hotspots = HashSet::new();
        
        for (i, node) in func.body.iter().enumerate() {
            match node {
                LairNode::Call { function, .. } => {
                    let count = self.call_counts.entry(function.clone()).or_insert(0);
                    *count += 1;
                    if *count > 10 {
                        hotspots.insert(i);
                    }
                }
                LairNode::Loop { .. } => {
                    let count = self.loop_iterations.entry(i).or_insert(0);
                    *count += 1;
                    if *count > 100 {
                        hotspots.insert(i);
                    }
                }
                _ => {}
            }
        }
        
        hotspots
    }
}
```

```rust
use bonsai_lair::LairFunction;

/// A collection of optimization passes.
pub struct PassManager {
    passes: Vec<Box<dyn OptimizationPass>>,
}

impl Default for PassManager {
    fn default() -> Self {
        Self {
            passes: vec![
                Box::new(ConstantFolding),
                Box::new(DeadCodeElimination),
                Box::new(CommonSubexpressionElimination),
                Box::new(LoopUnrolling { factor: 4 }),
            ],
        }
    }
}

impl PassManager {
    pub fn run(&self, func: &mut LairFunction) {
        for pass in &self.passes {
            pass.apply(func);
        }
    }
}

pub trait OptimizationPass: Send + Sync {
    fn name(&self) -> &str;
    fn apply(&self, func: &mut LairFunction);
}

pub struct ConstantFolding;
impl OptimizationPass for ConstantFolding {
    fn name(&self) -> &str { "constant_folding" }
    fn apply(&self, func: &mut LairFunction) {
        // Evaluate constant expressions at compile time
        for node in &mut func.body {
            if let bonsai_lair::LairNode::Return { value } = node {
                // If value is a constant expression, compute it
                if let Some(v) = value {
                    if v.parse::<i64>().is_ok() {
                        // Already a constant — nothing to fold
                    }
                }
            }
        }
    }
}

pub struct DeadCodeElimination;
impl OptimizationPass for DeadCodeElimination {
    fn name(&self) -> &str { "dead_code_elimination" }
    fn apply(&self, func: &mut LairFunction) {
        // Remove unreachable code after unconditional returns
        let mut keep = Vec::new();
        let mut returned = false;
        for node in &func.body {
            if returned {
                continue;
            }
            if matches!(node, bonsai_lair::LairNode::Return { .. }) {
                returned = true;
            }
            keep.push(node.clone());
        }
        func.body = keep;
    }
}

pub struct CommonSubexpressionElimination;
impl OptimizationPass for CommonSubexpressionElimination {
    fn name(&self) -> &str { "cse" }
    fn apply(&self, _func: &mut LairFunction) {
        // Identify and eliminate duplicate computations
    }
}

pub struct LoopUnrolling {
    pub factor: u32,
}
impl OptimizationPass for LoopUnrolling {
    fn name(&self) -> &str { "loop_unrolling" }
    fn apply(&self, func: &mut LairFunction) {
        // Unroll loops by the given factor
        let mut new_body = Vec::new();
        for node in &func.body {
            if let bonsai_lair::LairNode::Loop { body } = node {
                for _ in 0..self.factor {
                    new_body.extend(body.clone());
                }
            } else {
                new_body.push(node.clone());
            }
        }
        func.body = new_body;
    }
}
```

```rust
        }

        CacheStats {
            entries: self.entries.len(),
            capacity: self.max_entries,
        }
    }
```
