```bash

cargo build --release --bin bonsai-bot-main
```

```powershell

./scripts/bonsai-bot-cli.ps1 -Command submit -Task "Run full ecosystem pipeline"

# Execute pending tasks
./scripts/bonsai-bot-cli.ps1 -Command execute

# Check status
./scripts/bonsai-bot-cli.ps1 -Command status

# Interactive mode
./scripts/bonsai-bot-cli.ps1 -Command interactive
```

### Production Deployment

```powershell
# Build release binary
cargo build --release --bin bonsai-bot-main

          -Destination "C:\BonsaiBot\bonsai-bot.exe"

              -ArgumentList "interactive"
```

### Docker Deployment

```dockerfile
FROM rust:latest
WORKDIR /app
COPY . .
RUN cargo build --release --bin bonsai-bot-main
CMD ["./target/release/bonsai-bot-main", "interactive"]
```
