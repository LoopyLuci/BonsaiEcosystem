`crates/bonsai-regex-frontend/src/lib.rs`

```rust
use anyhow::Result;
use bonsai_lair::{LairModule, LairFunction, FunctionSignature, BuirType, Effect};
use bonsai_language_frontend::LanguageFrontend;
use regex::Regex;
use std::path::Path;

pub struct RegexFrontend {
    name: String,
    extensions: Vec<String>,
    function_patterns: Vec<Regex>,
    comment_patterns: Vec<Regex>,
}

impl RegexFrontend {
    pub fn new(name: &str, extensions: Vec<String>) -> Self {
        let function_patterns = Self::get_function_patterns(name);
        let comment_patterns = Self::get_comment_patterns(name);
        Self {
            name: name.to_string(),
            extensions,
            function_patterns,
            comment_patterns,
        }
    }

    fn get_function_patterns(lang: &str) -> Vec<Regex> {
        let patterns: Vec<&str> = match lang.to_lowercase().as_str() {
            // C-like languages
            "c" | "c++" | "c#" | "java" | "javascript" | "typescript" | "go" | "rust" |
            "swift" | "kotlin" | "dart" | "scala" | "zig" | "nim" | "d" => vec![
                r"(?:func|fn|fun|function|def|sub|void|int|bool|string|var|let|const|public|private|static)\s+(\w+)\s*\(",
                r"(\w+)\s*:\s*\(.*?\)\s*->",
            ],
            // Python-like
            "python" | "ruby" | "julia" | "r" | "lua" => vec![
                r"def\s+(\w+)\s*\(",
                r"async\s+def\s+(\w+)\s*\(",
                r"class\s+(\w+)",
            ],
            // Lisp-like
            "clojure" | "common lisp" | "scheme" | "racket" | "elisp" => vec![
                r"\(\s*defun\s+(\w+)",
                r"\(\s*define\s*\(\s*(\w+)",
                r"\(\s*defn\s+(\w+)",
                r"\(\s*defmacro\s+(\w+)",
            ],
            // ML-like
            "haskell" | "ocaml" | "f#" | "elm" | "purescript" | "gleam" | "roc" => vec![
                r"(\w+)\s*::\s*",
                r"let\s+(\w+)\s*=",
                r"(\w+)\s*:\s*.*?=",
            ],
            // Shell
            "shell" | "bash" | "zsh" | "perl" | "awk" | "tcl" | "rexx" => vec![
                r"function\s+(\w+)\s*\{",
                r"(\w+)\s*\(\s*\)\s*\{",
                r"sub\s+(\w+)\s*\{",
            ],
            // Esoteric languages
            "brainfuck" => vec![r"(.{1,20})"], // Chunks of instructions
            "lolcode" => vec![r"HAI\s*\n.*?KTHXBYE"],
            "shakespeare" => vec![r"Act\s+\w+.*?Scene\s+\w+"],
            "chef" => vec![r"(\w+\s+){1,5}\."],
            // Fallback: any word followed by brackets
            _ => vec![
                r"(\w+)\s*\(",
                r"(\w+)\s*:=",
                r"(\w+)\s*=",
            ],
        };
        patterns.iter().map(|p| Regex::new(p).unwrap()).collect()
    }

    fn get_comment_patterns(_lang: &str) -> Vec<Regex> {
        vec![
            Regex::new(r"//.*").unwrap(),
            Regex::new(r"#.*").unwrap(),
            Regex::new(r"--.*").unwrap(),
            Regex::new(r";.*").unwrap(),
            Regex::new(r"\(\*.*?\*\)").unwrap(),
            Regex::new(r"/\*.*?\*/").unwrap(),
        ]
    }

    fn clean_source(&self, source: &str) -> String {
        let mut cleaned = source.to_string();
        for pattern in &self.comment_patterns {
            cleaned = pattern.replace_all(&cleaned, "").to_string();
        }
        cleaned
    }
}

#[async_trait::async_trait]
impl LanguageFrontend for RegexFrontend {
    fn language_name(&self) -> &str { &self.name }
    fn file_extensions(&self) -> &[&str] {
        self.extensions.iter().map(|s| s.as_str()).collect::<Vec<_>>().leak()
    }

    async fn parse(&self, source: &str, _path: &Path) -> Result<LairModule> {
        let cleaned = self.clean_source(source);
        let mut functions = Vec::new();
        let mut seen = std::collections::HashSet::new();

        for pattern in &self.function_patterns {
            for cap in pattern.captures_iter(&cleaned) {
                if let Some(name) = cap.get(1) {
                    let func_name = name.as_str().to_string();
                    if seen.insert(func_name.clone()) && !func_name.len() > 50 {
                        functions.push(LairFunction {
                            name: func_name,
                            signature: FunctionSignature { params: vec![], return_type: BuirType::Dynamic },
                            body: vec![],
                            effects: vec![Effect::Io],
                            metadata: Default::default(),
                        });
                    }
                }
            }
        }

        Ok(LairModule {
            name: format!("{}_module", self.name.to_lowercase()),
            language: self.name.clone(),
            version: "0.1.0".into(),
            functions,
            types: vec![],
            imports: vec![],
            exports: vec![],
        })
    }
}
```

### Step 2 – Generator Tool (`lang-gen`)

```rust
// lang-gen/src/main.rs
use std::fs;
use std::path::Path;
use serde::{Serialize, Deserialize};

#[derive(Debug, Deserialize)]
struct Language {
    name: String,
    extensions: Vec<String>,
    parser: Option<String>,
}

#[derive(Debug, Deserialize)]
struct Config {
    languages: Vec<Language>,
}

fn main() {
    let config_yaml = fs::read_to_string("languages.yaml").unwrap();
    let config: Config = serde_yaml::from_str(&config_yaml).unwrap();
    
    for lang in config.languages {
        let crate_name = format!("omnisystem-{}", lang.name.to_lowercase());
        let crate_path = Path::new("crates").join(&crate_name);
        fs::create_dir_all(&crate_path).unwrap();
        
        let cargo_toml = format!(
            r#"[package]
name = "{}"
version = "0.1.0"
edition = "2021"

[dependencies]
anyhow = "1"
bonsai-lair = {{ path = "../bonsai-lair" }}
bonsai-language-frontend = {{ path = "../bonsai-language-frontend" }}
"#,
            crate_name
        );
        fs::write(crate_path.join("Cargo.toml"), cargo_toml).unwrap();
        
        let lib_rs = format!(
            r#"use bonsai_language_frontend::LanguageFrontend;
use frontend::{}Frontend;

mod frontend;

inventory::submit! {{
    bonsai_language_frontend::LanguageRegistration {{
        name: "{}",
        factory: || Box::new({}Frontend::new()),
    }}
}}
"#,
            lang.name.replace(" ", ""),
            lang.name,
            lang.name.replace(" ", "")
        );
        fs::write(crate_path.join("src/lib.rs"), lib_rs).unwrap();
        
        let frontend_rs = format!(
            r#"use bonsai_language_frontend::LanguageFrontend;
use bonsai_lair::*;
use std::path::Path;
use std::process::Command;

pub struct {}Frontend;

impl {}Frontend {{
    pub fn new() -> Self {{ Self }}
}}

    fn language_name(&self) -> &str {{ "{}" }}
    fn file_extensions(&self) -> &[&str] {{ &{:?} }}
    
    async fn parse(&self, source: &str, file_path: &Path) -> anyhow::Result<LairModule> {{
        // Placeholder: In a real implementation, you would call the external parser.
        // For {}: {}
        Ok(LairModule {{
            name: file_path.file_stem().unwrap().to_string_lossy().to_string(),
            language: "{}".into(),
            version: "0.1.0".into(),
            functions: vec![],
            types: vec![],
            imports: vec![],
            exports: vec![],
        }})
    }}
}}
"#,
            lang.name.replace(" ", ""),
            lang.name.replace(" ", ""),
            lang.name.replace(" ", ""),
            lang.name,
            lang.extensions,
```

## 🌐 How to Add Any Other Language

1. **Add an entry** to `languages.yaml` with the language name, file extensions, and an optional external parser command (e.g., compiler invocation that outputs a JSON AST).
2. **Run the generator** – the crate is created automatically.
3. **Build** – `cargo build --workspace`.
4. **The language is now supported** – the `LanguageRouter` will detect it by file extension and can produce a LAIR module (stub; real parsing requires implementing the external parser integration).
